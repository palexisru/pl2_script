Структура файлов базы данных

System \ Основной файл \
    scheme_name С(32) \ имя схемы данных \
    parent_name С(32) \ имя родительской схемы данных \
    code_page C(32)   \ кодовая страница по умолчанию \
    scheme_tuple T    \ уникальный идентификатор схемы \
    last_tuple Int32  \ номер последнего идентификатора, использованного для генерации универсального ключа схемы \

Schema_list \ список известных схем \
    schema_name C(32)  \ имя схемы \
    schema_path C(254) \ URL схемы \
    schema_mode        \ режим \
        parent - родительская схема для данной схемы
        chield - дочерняя схема данной схемы
        external - внешняя схема
    read_only L        \ разрешен доступ только на чтение \
    scheme_tuple T \ уникальный идентификатор схемы \
   
Schema_meta \ метаописание объектов схемы \
    table_Name C(32) \ имя таблицы \
    field_name C(32) \ имя поля \
    field_pos  Int32 \ позиция в файле \ 
    field_type       \ тип данных \
        L logical     \ C(1) T,F )
        I integer     \ Int32 0xFFFFFFFF \
        R real        \ float32 (decimal = 8) 0xFFFFFFFF \
        N numeric     \ ( len <=8; decimal <= len-2 ) \
        C string      \ ( len <= 234 ) \
        D date        \ 8+6+1 = yyyymmddhhmmssd \
        T tuple       \ Int32 0xFFFFFFFF уникальный идентификатор в схеме \
        K keep (D)    \ дата/время пометки кортежа на удаление \
    field_Len  Int32 \ длина поля \
    field_dec  Int32 \ количество десятичных знаков \
    caption    C(32) \ заголовок \
    section    C(32) \ раздел вида информации \
    comment    C(254)\ описание поля \
    auto_incriment L \ автоинкримент последнего значения \
    look_table C(32) \ таблица для ссылки \
    look_field C(32) \ поле для поиска значения \
    display_field C(32) \ поле для отображения значения \
    look_scheme C(32) \ схема размещения таблицы \

Scheme_references \ связи между кортежами схемы \
    parent_tuple  T \ родительский идентификатор \
    chield_tuple  T \ связанный идентиикатор \

External_references \ связи с кортежами других схем \    
    parent_scheme T \ идентификатор родительской схемы из Schema_list\
    parent_tuple  T \ родительский идентификатор \
    chield_scheme T \ идентификатор дочерней схемы из Schema_list\
    chield_tuple  T \ связанный идентиикатор \

Scheme_users    \ список пользователей \
    user_id   T     \ идентификатор в схеме \
    user_name C(32) \ имя пользователя \
    password  C(32) \ пароль \
    active    L     \ признак возможности использования \

Scheme_right       \ список прав пользователей на объекты \
    user_id            \ уникальный идентификатор пользователя \
    table_name         \ имя таблицы \
    field_name         \ имя поля \
    user_name          \ имя пользователя \
    access_right C (6) \ права пользователя \
    object_type  C (1) \ вид объекта \
        S - scheme
        T - table
        F - field
        U - user

Access_list \ флаги разрешений доступа \
    F \ разрешен поиск объектов \
    f \ запрещен поиск объектов \
    R \ разрешен просмотр объектов \
    r \ запрещен просмотр объектов \
    W \ разрешено изменение объектов \
    w \ запрещено изменение объектов \
    C \ разрешено создание объектов \
    c \ запрещено создание объектов \
    D \ разрешено удаление объектов \
    d \ запрещено удаление объектов \
    M \ разрешено управление правами доступа \
    m \ запрещено управление правами доступа \
